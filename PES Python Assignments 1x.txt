def biggestAndSmallest():
    try:
        N = int(input("Enter the range of numbers you want to enter: "))
        for i in range(N):
            number = float(input("Enter your number: "))
            yield number
    except ValueError as e:
        print("Invald number entered!")


maximum = float('-inf')
minimum = float('inf')
for number in biggestAndSmallest():
    if number>maximum:
        maximum = number
    if number<minimum:
        minimum = number
print("maximum: {},minimum: {}".format(maximum,minimum))
